Data Preparation Script (example)


import pandas as pd

# Load Q2 data
df = pd.read_csv("q2_revenue.csv")

# Clean and compute growth
df['revenue_prev_q'] = df['revenue'].shift(1)
df['qoq_growth_pct'] = (df['revenue'] - df['revenue_prev_q']) / df['revenue_prev_q'] * 100
print(df[['segment','revenue','qoq_growth_pct']].head())
